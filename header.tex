\def\incrementbar#1{%
  \b@rnob@ck=#1%
  \advance\b@rnob@ck by -1%
  \advance\b@rnob@ck by \barno%
  \barno=\b@rnob@ck%
}%
\def\s@r{\vrule\@width.6\p@\@height1\h@lf\internote\@depth1\h@lf\internote}%
\def\overlong{%
    \s@r
    \vrule\@width8.33\elemskip\@height\h@lf\internote\@depth\h@lf\internote
    \s@r}
\def\longrest#1{%
  \ifnum#1>4
    \znotes\uptext{\lower3\Interligne\hbox{\overlong}}\enotes
    \znotes\uptext{\lower7\Interligne\hbox to 8.66\elemskip{\hss{\bigtype#1}\hss}}\enotes
    \addspace{10\elemskip}%
  \else\ifnum#1=4
    \znotes\uptext{\lower7\Interligne\hbox to 8.00\elemskip{\hss{\bigtype#1}\hss}}\enotes
    \Notes\hqsk\sk\PAUSe\sk\enotes
	%addspace{7.5\elemskip}%
	%def\atnextbar{\znotes\centerbar{\PAUSe}\enotes}%
  \else\ifnum#1=3
    \znotes\uptext{\lower7\Interligne\hbox to 6.66\elemskip{\hss{\bigtype#1}\hss}}\enotes
    \notes\sk\PAuse\pause\sk\enotes
	%addspace{7.0\elemskip}%
	%def\atnextbar{\znotes\centerbar{\PAuse\qsk\pause}\enotes}%
  \else\ifnum#1=2
    \znotes\uptext{\lower7\Interligne\hbox to 4.66\elemskip{\hss{\bigtype#1}\hss}}\enotes
    \notes\sk\PAuse\sk\enotes
	%addspace{6.5\elemskip}%
	%def\atnextbar{\znotes\centerbar{\PAuse}\enotes}%
  \else%ifnum#1=1
    \znotes\uptext{\lower7\Interligne\hbox to 4.8\elemskip{\hss{\bigtype#1}\hss}}\enotes
    \NOtes\hsk\pause\enotes
	%addspace{6.0\elemskip}%
	%def\atnextbar{\znotes\centerpause\enotes}%
  \fi\fi\fi\fi
  \incrementbar{#1}%
  }
\newdimen\interb@rswidth
\newdimen\restdispl@ywidth
\newdimen\p@useb@rwidth
\def\longlongrest#1#2{%
  \interb@rswidth=#2\elemskip
  \restdispl@ywidth=#2\elemskip
  \ifnum#1>4
    \advance\restdispl@ywidth by 8.66\elemskip
    \advance\interb@rswidth by 10\elemskip
    \p@useb@rwidth=\restdispl@ywidth
    \advance\p@useb@rwidth by -0.33\elemskip
    \znotes\uptext{\lower3\Interligne\hbox{\s@r\vrule\@width\p@useb@rwidth\@height\h@lf\internote\@depth\h@lf\internote\s@r}}\enotes
  \else\ifnum#1=4
    \advance\restdispl@ywidth by 8.33\elemskip
    \advance\interb@rswidth by 10\elemskip
	\def\atnextbar{\znotes\centerbar{\PAUSe}\enotes}%
  \else\ifnum#1=3
    \advance\restdispl@ywidth by 6.66\elemskip
    \advance\interb@rswidth by 8.33\elemskip
	\def\atnextbar{\znotes\centerbar{\loff{\PAuse}\roff{\pause}}\enotes}%
  \else\ifnum#1=2
    \advance\restdispl@ywidth by 4.66\elemskip
    \advance\interb@rswidth by 5.3\elemskip
	\def\atnextbar{\znotes\centerbar{\PAuse}\enotes}%
  \else%ifnum#1=1
    \advance\restdispl@ywidth by 4.80\elemskip
    \advance\interb@rswidth by 6.7\elemskip
	\def\atnextbar{\znotes\centerbar{\pause}\enotes}%
  \fi\fi\fi\fi
  \znotes\uptext{\lower7\Interligne\hbox to \restdispl@ywidth{\hss{\bigtype#1}\hss}}\enotes
  \addspace\interb@rswidth%
  \incrementbar{#1}%
}%
\def\tacet#1#2{%
  \p@useb@rwidth=#2\elemskip
  \interb@rswidth=#2\elemskip
  \restdispl@ywidth=#2\elemskip
  \advance\p@useb@rwidth by 8.33\elemskip
  \advance\restdispl@ywidth by 8.66\elemskip
  \advance\interb@rswidth by 10\elemskip
  \znotes\uptext{\lower3\Interligne\hbox{\s@r\vrule\@width\p@useb@rwidth\@height\h@lf\internote\@depth\h@lf\internote\s@r}}\enotes
  \znotes\uptext{\lower8\Interligne\hbox to\restdispl@ywidth{\hss{\bigtype Tacet}\hss}}\enotes
  \addspace\interb@rswidth%
  \incrementbar{#1}%
}%
