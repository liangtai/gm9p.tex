\def\longrestw#1{%
  \ifnum#1>4
    \znotes\uptext{\lower3\Interligne\hbox{\overlong}}\nextstaff\uptext{\lower3\Interligne\hbox{\overlong}}\enotes%
    \znotes\zmidstaff{\hbox to 8.66\elemskip{\hss{\bigtype#1}\hss}}\enotes%
    \addspace{10\elemskip}%
  \else\ifnum#1=4
    \znotes\zmidstaff{\hbox to 7.66\elemskip{\hss{\bigtype#1}\hss}}\enotes%
    \Notes\sk\PAUSe\sk\nextstaff\sk\PAUSe\sk\enotes%
  \else\ifnum#1=3%
    \znotes\zmidstaff{\hbox to 6.66\elemskip{\hss{\bigtype#1}\hss}}\enotes%
    \notes\sk\PAuse\pause\sk\nextstaff\sk\PAuse\pause\sk\enotes%
  \else\ifnum#1=2
    \znotes\zmidstaff{\hbox to 4.66\elemskip{\hss{\bigtype#1}\hss}}\enotes%
    \notes\sk\PAuse\sk\nextstaff\sk\PAuse\sk\enotes%
  \else%ifnum#1=1
    \NOtes\hsk\pause\nextstaff\hsk\pause\enotes%
  \fi\fi\fi\fi
  \incrementbar{#1}%
  }
